 10          """ 
 11          Generate a random graph using the local attachment model. 
 12           
 13          **Args**: 
 14           
 15                  * ``n`` (int): the number of nodes to add to the graph 
 16                  * ``m`` (int): the number of edges a new node will add to the graph 
 17                  * ``r`` (int): the number of edges (of the ``m``) that a node will add to uniformly selected random nodes. 
 18                    All others will be added to neighbors of the ``r`` selected nodes. 
 19           
 20          **KwArgs**: 
 21                  * ``seed [=-1]`` (int): a seed for the random number generator 
 22                  * ``graph [=None]`` (:py:class:`zen.DiGraph`): the graph that will be populated.  If the graph is ``None``,  
 23                    then a new graph will be created. 
 24           
 25          **Returns**: 
 26                  :py:class:`zen.DiGraph`. The graph generated. 
 27           
 28          .. note:: 
 29                  Source: M. O. Jackson and B. O. Rogers "Meeting strangers and friends of friends: How random are social networks?" The American Economic Review, 2007. 
 30          """ 
 31          seed = kwargs.pop('seed',None) 
 32          graph = kwargs.pop('graph',None) 
 33

 35                  raise ZenException, 'The graph provided must be directed' 
 36          if graph is not None and len(graph) > 0: 
 37                  raise ZenException, 'The graph provided is not empty' 
 38

 40                  raise ZenException, 'Unknown arguments: %s' % ', '.join(kwargs.keys())

 43                  raise ZenException, 'r must be an integer' 
 44          elif r < 1:

 54          # build the initial graph 
 55

 57          if G is None:

 61          for i in range(m+1):

 63                   
 64          # according to Jackson's paper, all initial nodes have m neighbors. 
 65          for i in range(m+1): 
 66                  for j in range(m+1): 
 67                          if j != i: 
 68                                  G.add_edge(j,i)

 70          ###### 
 71          # Build the rest of the graph 
 72          node_list = list(range(m+1)) 
 73          for i in range(m+1,n): 
 74                  G.add_node(i) 
 75                   
 76                  # pick random neighbors (the parents) 
 77                  parents = random.sample(node_list,r) 
 78                   
 79                  # pick neighbors from the parents' neighborhoods 
 80                  potentials = set() 
 81                  for n in parents: 
 82                          potentials.update(G.out_neighbors(n)) 
 83                           
 84                  potentials.difference_update(parents) 
 85                  nsize = min([m-r,len(potentials)]) 
 86                  neighbors = random.sample(potentials,nsize) 
 87                   
 88                  # connect 
 89                  for v in (parents + neighbors): 
 90                          G.add_edge(i,v) 
 91                           
 92                  node_list.append(i) 
 93                           
 94          # done 
 95          return G
